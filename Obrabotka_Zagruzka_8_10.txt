Процедура ПутьКФайлуНачалоВыбора(Элемент, ДанныеВыбора, СтандартнаяОбработка)
	СтандартнаяОбработка = Ложь;
	ДиалогВыбора = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.Открытие);
	ДиалогВыбора.МножественныйВыбор = Ложь;
	ДиалогВыбора.Заголовок = "Выбрать файл";
	Если ДиалогВыбора.Выбрать() Тогда
		ПутьКФайлу = ДиалогВыбора.ПолноеИмяФайла; 
		РасширениеФайла = Прав(ПутьКФайлу, 4);
		Если Лев(РасширениеФайла, 1) = "." Тогда
			РасширениеФайла = Прав(РасширениеФайла, 3);
		КонецЕсли;
	КонецЕсли;
КонецПроцедуры

Процедура ЗагрузитьДанныеНаСервереCSV()
	Текст = Новый ТекстовыйДокумент;
	Текст.Прочитать(ПутьКФайлу);
	//Индекс=1;
	Для Индекс=2 по Текст.КоличествоСтрок() Цикл
		Стр = Текст.ПолучитьСтроку(Индекс) ;
		//1. Разбор на поля одной строки
		МассивПолей=СтрРазделить (Стр, ";",Истина);
		ID_туриста=СокрЛП(МассивПолей[0]);
		Фамилия_Имя_Отчество=СокрЛП(МассивПолей[1]);
		Пол=СокрЛП(МассивПолей [2]);
		Дата_рождения=СокрЛП(МассивПолей[3]);
		Группа_По_Осн_месту_работы=СокрЛП(Массив
		Какие_Регионы_Стран_Привлекательны_ДляТуризма=СокрЛП(МассивПолей[5]);
		НаличиеЗагранпаспорта=СокрЛП(МассивПолей[6]);
		Рост=СокрЛП(МассивПолей[7]);
		ПредпочтительныйВидТуризма=СокрЛП(МассивПолей[8]);
		// 2. Ищем объект, если есть - открываем по ссылке на запись, если нет - создаем новый
		СправочникСсылка = Справочники.Туристы.НайтиПоКоду(ID_туриста, Истина);
		Если СправочникСсылка.Пустая() Тогда
			// если не нашли, значит надо добавить новую запись в справочник.
			Сообщить ("Будет создан элемент справочника Туристы с кодом: " + ID_туриста) ;
			СправочникОбъект = Справочники.Туристы.СоздатьЭлемент();
			СправочникОбъект.Код = ID_туриста;
		Иначе
			// если нашли, то значит надо получить объект на изменение
			СправочникОбъект = СправочникСсылка.ПолучитьОбъект ();
		КонецЕсли;
		// Строковые реквизиты можно записывать без преобразования
		СправочникОбъект.Наименование=Фамилия_Имя_Отчество;
		// записать реквизит Рост, преобразовав в Число
		Попытка
			СправочникОбъект.Рост=Число(Рост);
		Исключение
			Сообщить ("Строка №"+Индекс+". Не удалось привести Рост к числу");
		КонецПопытки;

        // преобразование поля типа Дата
		Если Не Дата_рождения="" Тогда
			День=Число(Лев(Дата_рождения, 2));
			Месяц=Число(Сред(Дата_рождения, 4, 2));
			Год=Число(Прав(Дата_рождения, 4));
			СправочникОбъект.ДатаРождения=Дата(Год, Месяц, День);
		КонецЕсли;
		// и так далее все поля преобразовываем к нужному типу,
		// как делали это при загрузке из табл. документа
		Если НаличиеЗагранпаспорта="" Тогда
		ИначеЕсли НаличиеЗагранпаспорта="-" Тогда
		СправочникОбъект.НаличиеЗагранпаспорта=Истина;
		СправочникОбъект.НаличиеЗагранпаспорта=Ложь;
		КонецЕсли;
		Если ВРег(Лев(Пол, 1))="М" Тогда
			СправочникОбъект.Пол=Перечисления.Пол.Мужской;
		ИначеЕсли ВРег (Лев (Пол, 1))="Ж" Тогда 
			СправочникОбъект.Пол=Перечисления.Пол.Женский;
		Иначе  
			Сообщить ("Строка№"+Индекс+". Не удалось сопоставить пол"); 
			Продолжить;
		КонецЕсли;
	
		ГруппаСсылка=Справочники.ГруппаПоОснМестуРаботы.НайтиПоНаименованию(Группа_По_Осн_месту_работы);
		// поле Группа По Осн месту работы
		Если ГруппаСсылка.Пустая() Тогда
			Сообщить ("Строка!"+Индекс+". Не найдена группа с наименованием "+Группа_По_Осн_месту_работы) ;
			ЭлементГруппы=Справочники.ГруппаПоОснМестуРаботы.СоздатьЭлемент();
			ЭлементГруппы.Наименование=Группа_По_Осн_месту_работы;
			ЭлементГруппы.Записать();
			ГруппаСсылка=ЭлементГруппы.Ссылка;
		КонецЕсли;
		СправочникОбъект.ГруппаПоОснМестуРаботы=ГруппаСсылка;
		//заполнение табличной части Какие Регионы Стран Привлекательны ДляТуризма
		ПредпочтительныеГруппыМассив=СтрРазделить(СокрЛП(Какие_Регионы_Стран_Привлекательны_ДляТуризма), ",");
		СправочникОбъект.Какие_Регионы_Стран_Привлекательны_ДляТуризма.Очистить ();
		Для каждого ОтдельныйРегион из ПредпочтительныеГруппыМассив Цикл
			Попытка
				РегионСсылка=Справочники.РегионыСтран.НайтиПоНаименованию(СокрЛП(ОтдельныйРегион));
			Исключение
				Сообщить ("Строка!?"+Индекс+". Не найден регион с наименованием "+ОтдельныйРегион) ;
			КонецПопытки;
			
			Если Не РегионСсылка.Пустая() Тогда
				НовСтрока=СправочникОбъект.Какие_Регионы_Стран_Привлекательны_ДляТуризма.Добавить() ;
				НовСтрока.ПривлекательныеРегионыСтран=РегионСсылка;
			Иначе
				Сообщить ("Строкаl?"+Индекс+". Не найден регион с наименованием "+ОтдельныйРегион);
				ЭлементРегион=Справочники.РегионыСтран.СоздатьЭлемент ();
				ЭлементРегион.Наименование=СокрЛП(ОтдельныйРегион) ;
				ЭлементРегион.Записать();
				РегионСсылка=ЭлементРегион.Ссылка;
				НовСтрока=СправочникОбъект.Какие_Регионы_Стран_Привлекательны_ДляТуризма.Добавить();
				НовСтрока.ПривлекательныеРегионыСтран=РегионСсылка;
			КонецЕсли;
		КонецЦикла;
	
		//заполнение табличной части ПредпочтительныйВидТуризма
		ПредпочтительныеВидыТуризмаМассив=СтрРазделить(СокрЛП(ПредпочтительныйВидТуризма), ",");

			Попытка
				ВидТуризмаСсылка=Справочники.ВидыТуризма.НайтиПоНаименованию(СокрЛП(ОтдельныйВидТуризма));

Процедура ЗагрузитьДанныеНаСервереXLSX()

	ТабличныйДокумент = Новый ТабличныйДокумент;

		Область = Неопределено;
	// 3) Ищем нужную страницу в считанном табличном документе
	Попытка
		Область = ТабличныйДокумент.Область("Туристы");

Процедура ЗагрузитьДанные(Команда)
	Отказ = Ложь;
	Если Не ЗначениеЗаполнено(ПутьКФайлу) Тогда
		Сообщить("Путь к файлу отсутствует!");
		Отказ = Истина;
	КонецЕсли;
	Если Отказ Тогда
		ПоказатьПредупреждение(, "Не удалось загрузить файл!");
	Иначе
		Если РасширениеФайла = "xlsx" или РасширениеФайла = "xls" Тогда
			ЗагрузитьДанныеНаСервереXLSX();
		ИначеЕсли РасширениеФайла =  "csv" Тогда
			ЗагрузитьДанныеНаСервереCSV();
		КонецЕсли;	
	КонецЕсли;
